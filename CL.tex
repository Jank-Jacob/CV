%-------------------------------------------------------------------------------
% CONFIGURATIONS
%-------------------------------------------------------------------------------
\documentclass[11pt, a4paper]{awesome-cv}
\usepackage{csquotes}
\geometry{left=1.9cm, top=.5cm, right=1.9cm, bottom=1.4cm, footskip=.5cm}
% Highlight colour
\definecolor{awesome}{HTML}{00aeef}


% Set false if you don't want to highlight section with awesome color
\setbool{acvSectionColorHighlight}{true}

% ---------------------------------------------------------------------------
% Resolve cover letter folder.
% \clfolder is passed in via the xelatex -jobname invocation in build.sh.
% Falls back to CoverLetter/example if not defined.
% ---------------------------------------------------------------------------
\ifdefined\clfolder%
  \typeout{Coverletter: using folder '\clfolder'}
\else
  \def\clfolder{CoverLetter/example}
  \typeout{Coverletter: no folder specified, falling back to CoverLetter/example}
\fi

% Load job-specific details first (defines \companyName, \jobTitle, etc.)
\input{\clfolder/job_details.tex}

% Load shared config (identity, toggles — jobtype is set inside job_details)
\input{shared_configs.tex}

% ---------------------------------------------------------------------------
% Override awesome-cv defaults for tighter layout
% ---------------------------------------------------------------------------
\makeatletter
% 1. Unified Font Styles
\renewcommand*{\recipienttitlestyle}[1]{{\fontsize{10pt}{1.2em}\bodyfont\bfseries\color{black} #1}}
\renewcommand*{\recipientaddressstyle}[1]{{\fontsize{10pt}{1.2em}\bodyfont\color{black} #1}}
\renewcommand*{\letterdatestyle}[1]{{\fontsize{10pt}{1em}\bodyfont\itshape\color{black} #1\/}}
% 2. Redefine \makelettertitle for tight grouping
\renewcommand*{\makelettertitle}{%
  % Gap between CV Header and the Start of the Letter
  \vspace{6mm}
  % 1. The Date
  {\letterdatestyle{\@letterdate}\par}
  % 2. Minimal gap between Date and Company Name
  \nointerlineskip%
  \vspace{1mm}
  {\recipienttitlestyle{\@recipientname}\par}
  % 3. No gap between Company Name and Address lines
  \nointerlineskip%
  {\raggedright\recipientaddressstyle{\@recipientaddress}\par}
  % 4. Tight gap before the Subject Line
  \vspace{3mm}
  \ifthenelse{\isundefined{\@lettertitle}}{}
    {{\lettertitlestyle{\@lettertitle}\par}}
  % 5. Small gap before "Dear Hiring Manager"
  \vspace{2mm}
  % Salutation
  \lettertextstyle{\@letteropening}
}
% 3. Tight gap between "Sincerely," and name
\renewcommand*{\makeletterclosing}{%
  \vspace{2mm}
  \lettertextstyle{\@letterclosing}\\[2mm]
  \lettertextstyle{\@firstname\ \@lastname}
  \par
}
\makeatother

% Prevent words being broken or hyphenated across lines
\hyphenpenalty=10000
\exhyphenpenalty=10000

% Letter specific info
\recipient{\companyName}{\companyAddress}
\letterdate{\today}
\lettertitle{Application for \jobTitle}
\letteropening{Dear Hiring Manager,\\}
\letterclosing{Sincerely,}

\makecvfooter{\today}{Jacob Church · Cover Letter}{\thepage}

\begin{document}
\makecvheader[C]
\makelettertitle%
\begin{cvletter}
  \input{\clfolder/opening.tex}
  \input{\clfolder/evidence.tex}
  \input{\clfolder/closing.tex}
\end{cvletter}
\makeletterclosing%
\end{document}
