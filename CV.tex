% Load job-specific overrides if a job folder was specified via --job.
% \clfolder is passed from build.sh alongside \jobtype.
% This allows job_details.tex to override \jobtype, \projectorder,
% \experienceorder, \inventorfirst, \regionoverride, \addressoverride etc.
% For generic builds (no --job), this block is skipped entirely.
%-------------------------------------------------------------------------------
\ifdefined\clfolder%
  \IfFileExists{\clfolder/job_details.tex}{
    \typeout{CV-generator: loading job overrides from \clfolder/job_details.tex}
    % Preserve the jobtype passed from the command line (build.sh --cv flag)
    % job_details.tex may define \jobtype for its preferred variant, but the
    % explicit command-line value always wins.
    \ifdefined\jobtype\edef\jobtypebackup{\jobtype}\fi
    \input{\clfolder/job_details.tex}
    \ifdefined\jobtypebackup\edef\jobtype{\jobtypebackup}\fi
  }{}
\fi
