% ---------------------------------------------------------------------------
% Override identity fields for specific job contexts.
% job_details.tex can redefine \cvaddress, \cvmobile etc. before this file
% is loaded — but since awesome-cv uses \address etc. directly, we allow
% job_details to call \renewcommand{\cvaddressoverride}{...} and we apply
% them here.
%
% Supported overrides (define in job_details.tex if needed):
%   \def\regionoverride{au}        — adds AU working rights line to header
%   \def\addressoverride{...}      — replaces address in header
% ---------------------------------------------------------------------------
\ifdefined\regionoverride%
  \ifnum\pdfstrcmp{\regionoverride}{au}=0
    \ifdefined\addressoverride%
      \address{\addressoverride{} \\ NZ Citizen --- Full Australian Working Rights}
    \else
      \address{Kingston, Wellington, New Zealand \\ NZ Citizen --- Full Australian Working Rights}
    \fi
  \fi
  \ifnum\pdfstrcmp{\regionoverride}{nz}=0
    \ifdefined\addressoverride%
      \address{\addressoverride{} \\ Full NZ Working Rights}
    \else
      \address{Kingston, Wellington, New Zealand}
      % No annotation needed for NZ roles — address speaks for itself
    \fi
  \fi
\else
  \ifdefined\addressoverride%
    \address{\addressoverride}
  \fi
\fi

% ---------------------------------------------------------------------------
% TOGGLE DEFINITIONS
% Note: isEmbed intentionally also enables isSoft.
% Embedded builds include all software content, plus hardware-specific items.
% ---------------------------------------------------------------------------
\newtoggle{isMech}
\newtoggle{isElec}
\newtoggle{isEmbed}
\newtoggle{isSoft}

\togglefalse{isMech}
\togglefalse{isElec}
\togglefalse{isEmbed}
\togglefalse{isSoft}

\ifdefined\jobtype%
  \edef\expandedjobtype{\jobtype}
  \ifnum\pdfstrcmp{\expandedjobtype}{mech}=0       \toggletrue{isMech} \fi
  \ifnum\pdfstrcmp{\expandedjobtype}{electrical}=0  \toggletrue{isElec} \fi
  \ifnum\pdfstrcmp{\expandedjobtype}{embedded}=0
    \toggletrue{isEmbed}
    \toggletrue{isSoft}   % isEmbed implies isSoft
  \fi
  \ifnum\pdfstrcmp{\expandedjobtype}{software}=0    \toggletrue{isSoft} \fi
  \ifnum\pdfstrcmp{\expandedjobtype}{mechatronics}=0
    \toggletrue{isMech}
    \toggletrue{isElec}
    \toggletrue{isEmbed}
    \toggletrue{isSoft}
  \fi
\else
  % No jobtype defined — show everything (pipeline generic build)
  \toggletrue{isMech}
  \toggletrue{isElec}
  \toggletrue{isEmbed}
  \toggletrue{isSoft}
\fi

% ---------------------------------------------------------------------------
% POSITION LABEL
% isEmbed is checked before isSoft — since isEmbed also sets isSoft, without
% this ordering embedded builds would always resolve to "Software Engineer".
% ---------------------------------------------------------------------------
\position{%
  \iftoggle{isMech}{Mechatronics Engineer}{%
    \iftoggle{isElec}{Electrical Engineer}{%
      \iftoggle{isEmbed}{Embedded Software Engineer}{%
        \iftoggle{isSoft}{Software Engineer}{Mechatronics Engineer}%
      }%
    }%
  }%
}
